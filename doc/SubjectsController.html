def get_subjects
    subjects = Hash.new
    list_subjects = Subject.all
    @user.lessons.each{|le| subjects[le.subject_id]=list_subjects.find(le.subject_id).name}
    render json: subjects.as_json, status: :ok
end

def get_tutor_subjects

end
